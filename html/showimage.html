Przykładowy obrazek stworzony w PHP


        // Określenie wymiarów obrazka
        $image = imagecreate(500, 300);
        
        // Ustawienie koloru tła
        $background_color = imagecolorallocate($image, 0, 153, 0);
        
        // Ustawienie koloru obrazka
        $text_color = imagecolorallocate($image, 255, 255, 255);
        
        // Stworzenie obrazka zawierającego tekst
        imagestring($image, 5, 150, 100,  "Testowy glowny napis", $text_color);
        imagestring($image, 3, 160, 120,  "Testowy poboczny napis", $text_color);
        
        // Określenie typu obrazka
        header("Content-Type: image/png"); 
        
        //Wyświetlenie obrazka
        imagepng($image);

        //Zniszczenie obrazka
        imagedestroy($image);
